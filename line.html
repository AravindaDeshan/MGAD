Project Overview

A Line Following Robot is an autonomous robot that detects and follows a predefined path (usually a black line on a white surface or vice versa) using infrared (IR) sensors. This project implements a Proportional-Integral-Derivative (PID) Controller to improve the robot's accuracy, smoothness, and responsiveness while following the line.

Technology Stack

Arduino Uno TCRT5000 IR Sensors Motor Driver Actuators PID Control Microcontroller

Key Features

  • PID-Based Precision Control
  • Multi-Sensor Detection System
  • Adaptive Motor Control
  • Customizable & Scalable Design
  • Real-Time Debugging & Tuning
  • Practical Applications

void calculatePosition() { 
// Simplified position calculation for 2 sensors 
if (leftSensor == BLACK_LINE && rightSensor == NO_LINE) { 
lastPosition = -1000; // Far left 
} else if (rightSensor == BLACK_LINE && leftSensor == NO_LINE) { 
lastPosition = 1000;  // Far right 
} else if (leftSensor == BLACK_LINE && rightSensor == BLACK_LINE) { 
lastPosition = 0;     // Centered 
} else { 
// No line detected - maintain last position 
} 
} 
void calculatePID() { 
error = lastPosition; 
integral += error; 
integral = constrain(integral, -MAX_SUM_ERROR, MAX_SUM_ERROR); 
derivative = error - lastError; 
pidValue = (Kp * error) + (Ki * integral) + (Kd * derivative); 
pidValue = constrain(pidValue, -MAX_PID_VALUE, MAX_PID_VALUE); 
lastError = error; 
} 
void stopMotors() { 
digitalWrite(LEFT_MOTOR_IN1, LOW); 
digitalWrite(LEFT_MOTOR_IN2, LOW); 
analogWrite(LEFT_MOTOR_ENA, 0); 
digitalWrite(RIGHT_MOTOR_IN3, LOW); 
digitalWrite(RIGHT_MOTOR_IN4, LOW); 
analogWrite(RIGHT_MOTOR_ENB, 0); 
} 
void followLineWithPID() {
int leftSpeed = BASE_SPEED - pidValue; 
int rightSpeed = BASE_SPEED + pidValue; 
leftSpeed = constrain(leftSpeed, MIN_SPEED, MAX_SPEED); 
rightSpeed = constrain(rightSpeed, MIN_SPEED, MAX_SPEED); 
setMotorSpeeds(leftSpeed, rightSpeed); 
handleSpecialCases(); 
} 
void setMotorSpeeds(int leftSpeed, int rightSpeed) { 
if (leftSpeed >= 0) { 
digitalWrite(LEFT_MOTOR_IN1, HIGH); 
digitalWrite(LEFT_MOTOR_IN2, LOW); 
} else { 
digitalWrite(LEFT_MOTOR_IN1, LOW); 
digitalWrite(LEFT_MOTOR_IN2, HIGH); 
leftSpeed = -leftSpeed; 
} 
if (rightSpeed >= 0) { 
digitalWrite(RIGHT_MOTOR_IN3, HIGH); 
digitalWrite(RIGHT_MOTOR_IN4, LOW); 
} else { 
digitalWrite(RIGHT_MOTOR_IN3, LOW); 
digitalWrite(RIGHT_MOTOR_IN4, HIGH); 
rightSpeed = -rightSpeed; 
} 
analogWrite(LEFT_MOTOR_ENA, leftSpeed); 
analogWrite(RIGHT_MOTOR_ENB, rightSpeed); 
} 
void moveForward(int speed) { 
digitalWrite(LEFT_MOTOR_IN1, HIGH); 
digitalWrite(LEFT_MOTOR_IN2, LOW); 
analogWrite(LEFT_MOTOR_ENA, speed); 
digitalWrite(RIGHT_MOTOR_IN3, HIGH); 
digitalWrite(RIGHT_MOTOR_IN4, LOW); 
analogWrite(RIGHT_MOTOR_ENB, speed); 
} 
void handleSpecialCases() { 
if (leftSensor == BLACK_LINE && rightSensor == BLACK_LINE) { 
integral = 0; 
moveForward(BASE_SPEED); 
return; 
} 
if (leftSensor == NO_LINE && rightSensor == NO_LINE) { 
if (error < 0) { 
setMotorSpeeds(BASE_SPEED / 2, BASE_SPEED);  // Turn right slowly 
} else { 
setMotorSpeeds(BASE_SPEED, BASE_SPEED / 2);  // Turn left slowly 
} 
return; 
}
